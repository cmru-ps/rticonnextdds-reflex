275 // Convert (a,1,A)(b,2,B)(c,3,C) into ((a,1,A))((b,2,B))((c,3,C))
276 // Otherwise preprocessor will get confused due to commas.
277 // The PARENTHESIZE macro expands as follows:
278 //
279 // RTI_FILLER_0 Attr ---->
280 // RTI_FILLER_0 (a,1,A)(b,2,B)(c,3,C) ---->
281 // ((a,1,A)) RTI_FILLER_1 (b,2,B)(c,3,C) ---->
282 // ((a,1,A)) ((b,2,B)) RTI_FILLER_0 (c,3,C) ---->
283 // ((a,1,A)) ((b,2,B)) ((c,3,C)) RTI_FILLER_1 ---->
284 // BOOST_PP_CAT(((a,1,A)) ((b,2,B)) ((c,3,C)) RTI_FILLER_1, _END) ---->
285 // ((a,1,A)) ((b,2,B)) ((c,3,C)) RTI_FILLER_1_END ---->
286 // ((a,1,A)) ((b,2,B)) ((c,3,C)) ---->
287 // voila!!!
